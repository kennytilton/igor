since () {
  if [ "$#" -lt 1 ]; then
    echo "Usage: recent-files MINUTES [EXT ...]"
    echo "Example:"
    echo "  recent-files 10"
    echo "  recent-files 5 jpg"
    echo "  recent-files 15 jpg png mp4"
    return 1
  fi

  local minutes="$1"
  shift

  if ! [[ "$minutes" =~ ^[0-9]+$ ]]; then
    echo "Error: MINUTES must be a non-negative integer"
    return 1
  fi

  if [ "$#" -eq 0 ]; then
    # No extension filter
    find . -type f -mmin "-$minutes" -print
    return 0
  fi

  # Build extension filter
  local expr=()
  for ext in "$@"; do
    ext="${ext#.}"  # remove leading dot if supplied
    expr+=( -iname "*.${ext}" -o )
  done
  unset 'expr[${#expr[@]}-1]'  # remove trailing -o

  find . -type f \( "${expr[@]}" \) -mmin "-$minutes" -print
}

xfade() {
    if [ "$#" -ne 3 ]; then
        echo "Usage: xfade image1.png image2.png output.mp4"
        return 1
    fi

    # Logic for Timing:
    # 1. Image 1 needs to exist for (Hold Start + Transition) = 1 + 2 = 3s
    # 2. Image 2 needs to exist for (Transition + Hold Final) = 2 + 3 = 5s
    # 3. Offset is the point the transition begins = 1s
    # 4. Duration is the length of the transition = 2s

    ffmpeg -y -loop 1 -t 3 -i "$1" \
           -loop 1 -t 5 -i "$2" \
    -filter_complex \
    "[0:v]scale=trunc(iw/2)*2:trunc(ih/2)*2,format=yuv420p[v0]; \
     [1:v]scale=trunc(iw/2)*2:trunc(ih/2)*2,format=yuv420p[v1]; \
     [v0][v1]xfade=transition=fade:duration=2:offset=1,format=yuv420p" \
    -movflags +faststart "$3"
}

ftype-count() {
  [ "$#" -lt 1 ] && {
    echo "Usage: ftype-count EXT [EXT...]" >&2
    return 1
  }

  for ext in "$@"; do
    # strip leading dot if provided
    ext="${ext#.}"

    count=$(find . -type f -iname "*.${ext}" -print 2>/dev/null | wc -l)
    printf "%-10s %8d\n" "$ext" "$count"
  done
}
